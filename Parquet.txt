import pandas as pd

# 1. Load the January data
print("Loading January data...")
df = pd.read_parquet('2025-01.parquet')

# 2. See how many rows and columns
print(f"Rows: {len(df)}")
print(f"Columns: {len(df.columns)}")
print("\nColumn names:")
print(df.columns.tolist())

# 3. Look at the first 5 rows
print("\nFirst 5 rows of data:")
print(df.head())

# 4. Convert the strange timestamp to a readable date/time
df['readable_time'] = pd.to_datetime(df['timestamp'], unit='ms')
print("\nFirst 3 readable times:")
print(df['readable_time'].head(3))

# 5. Add a 'date' column (just the year-month-day)
df['date'] = df['readable_time'].dt.date
print("\nFirst 3 dates:")
print(df['date'].head(3))

# 6. Show a few rows with only the columns we care about
print("\nSample of price data (open, high, low, close):")
print(df[['readable_time', 'open', 'high', 'low', 'close']].head(5))

# 7. Basic statistics of prices
print("\nBasic price statistics for January:")
print(df[['open', 'high', 'low', 'close']].describe())
